use std::cell::RefCell;
use std::collections::BTreeMap;
use candid::{CandidType, Decode, Encode, Principal};
use serde::{Deserialize, Serialize};
use crate::types::*;

#[derive(CandidType, Clone, Debug, Serialize, Deserialize)]
pub struct ThreePoolState {
    pub config: PoolConfig,
    /// Raw token balances in native decimals (NOT normalized).
    pub balances: [u128; 3],
    /// LP token balances per user.
    pub lp_balances: BTreeMap<Principal, u128>,
    /// Total LP tokens in circulation.
    pub lp_total_supply: u128,
    /// Accumulated admin fees per token (in native decimals).
    pub admin_fees: [u128; 3],
    /// Whether the pool is paused.
    pub is_paused: bool,
    /// Whether the pool has been initialized with first liquidity.
    pub is_initialized: bool,
}

impl Default for ThreePoolState {
    fn default() -> Self {
        Self {
            config: PoolConfig {
                tokens: [
                    TokenConfig { ledger_id: Principal::anonymous(), symbol: String::new(), decimals: 0, precision_mul: 1 },
                    TokenConfig { ledger_id: Principal::anonymous(), symbol: String::new(), decimals: 0, precision_mul: 1 },
                    TokenConfig { ledger_id: Principal::anonymous(), symbol: String::new(), decimals: 0, precision_mul: 1 },
                ],
                initial_a: 100,
                future_a: 100,
                initial_a_time: 0,
                future_a_time: 0,
                swap_fee_bps: 4,
                admin_fee_bps: 5000,
                admin: Principal::anonymous(),
            },
            balances: [0; 3],
            lp_balances: BTreeMap::new(),
            lp_total_supply: 0,
            admin_fees: [0; 3],
            is_paused: false,
            is_initialized: false,
        }
    }
}

thread_local! {
    static STATE: RefCell<ThreePoolState> = RefCell::new(ThreePoolState::default());
}

pub fn mutate_state<F, R>(f: F) -> R
where
    F: FnOnce(&mut ThreePoolState) -> R,
{
    STATE.with(|s| f(&mut s.borrow_mut()))
}

pub fn read_state<F, R>(f: F) -> R
where
    F: FnOnce(&ThreePoolState) -> R,
{
    STATE.with(|s| f(&s.borrow()))
}

pub fn replace_state(new_state: ThreePoolState) {
    STATE.with(|s| {
        *s.borrow_mut() = new_state;
    });
}

pub fn save_to_stable_memory() {
    STATE.with(|s| {
        let state = s.borrow();
        let bytes = Encode!(&*state).expect("Failed to encode 3pool state");
        let len = bytes.len() as u64;

        let needed_pages = (len + 8 + 65535) / 65536;
        let current_pages = ic_cdk::api::stable::stable64_size();
        if needed_pages > current_pages {
            ic_cdk::api::stable::stable64_grow(needed_pages - current_pages)
                .expect("Failed to grow stable memory");
        }

        ic_cdk::api::stable::stable64_write(0, &len.to_le_bytes());
        ic_cdk::api::stable::stable64_write(8, &bytes);
    });
}

pub fn load_from_stable_memory() {
    let mut len_bytes = [0u8; 8];
    ic_cdk::api::stable::stable64_read(0, &mut len_bytes);
    let len = u64::from_le_bytes(len_bytes) as usize;

    if len == 0 { return; }

    let mut bytes = vec![0u8; len];
    ic_cdk::api::stable::stable64_read(8, &mut bytes);

    let state: ThreePoolState = Decode!(&bytes, ThreePoolState)
        .expect("Failed to decode 3pool state from stable memory");
    replace_state(state);
}
```

use ethnum::U256;

/// Returns the current amplification coefficient, linearly interpolating
/// if a ramp is in progress.
pub fn get_a(
    initial_a: u64,
    future_a: u64,
    initial_a_time: u64,
    future_a_time: u64,
    current_time: u64,
) -> u64 {
    if current_time >= future_a_time || initial_a_time == future_a_time {
        return future_a;
    }

    let elapsed = current_time.saturating_sub(initial_a_time);
    let duration = future_a_time - initial_a_time;

    if future_a > initial_a {
        initial_a + (future_a - initial_a) * elapsed / duration
    } else {
        initial_a - (initial_a - future_a) * elapsed / duration
    }
}
```

Direct port of Curve's Vyper `get_D`, using ethnum::U256:

```rust
const N_COINS_U256: U256 = U256::new(3);
const MAX_ITERATIONS: usize = 256;

/// Compute the StableSwap invariant D for 3 tokens.
/// `xp` must be normalized to 18-decimal precision.
/// `amp` is the raw amplification coefficient (not multiplied by N_COINS).
///
/// Returns None if Newton's method fails to converge.
pub fn get_d(xp: &[U256; 3], amp: u64) -> Option<U256> {
    let s: U256 = xp[0] + xp[1] + xp[2];
    if s == U256::ZERO {
        return Some(U256::ZERO);
    }

    let ann = U256::from(amp) * N_COINS_U256; // A * n

    let mut d = s;
    let mut d_prev: U256;

    for _ in 0..MAX_ITERATIONS {
        // D_P = D^(n+1) / (n^n * prod(xp))
        // Computed step-by-step to avoid overflow:
        // d_p = D
        // for each x: d_p = d_p * D / (x * N_COINS)
        let mut d_p = d;
        for x in xp.iter() {
            // d_p = d_p * d / (x * 3)
            d_p = d_p * d / (*x * N_COINS_U256);
        }

        d_prev = d;

        // D = (Ann * S + D_P * N) * D / ((Ann - 1) * D + (N + 1) * D_P)
        let numerator = (ann * s + d_p * N_COINS_U256) * d;
        let denominator = (ann - U256::ONE) * d + (N_COINS_U256 + U256::ONE) * d_p;
        d = numerator / denominator;

        // Check convergence
        let diff = if d > d_prev { d - d_prev } else { d_prev - d };
        if diff <= U256::ONE {
            return Some(d);
        }
    }

    None // Failed to converge
}
```

**Step 3: Implement get_y**

```rust
/// Given a swap where token `i` has new balance `x`, compute the new balance
/// of token `j` that maintains the invariant D.
///
/// `i`: input token index
/// `j`: output token index
/// `x`: new balance of token i (after adding swap input)
/// `xp`: current normalized balances (before swap)
/// `amp`: raw amplification coefficient
/// `d`: the invariant (from get_d)
pub fn get_y(i: usize, j: usize, x: U256, xp: &[U256; 3], amp: u64, d: U256) -> Option<U256> {
    assert!(i != j, "i == j");
    assert!(i < 3 && j < 3, "invalid index");

    let ann = U256::from(amp) * N_COINS_U256;

    // c = D^(n+1) / (n^n * prod(known_x) * Ann)
    // We build c incrementally:
    let mut c = d;
    let mut s_ = U256::ZERO;

    for k in 0..3 {
        let x_k = if k == i {
            x
        } else if k == j {
            continue; // skip j — that's what we're solving for
        } else {
            xp[k]
        };
        s_ += x_k;
        c = c * d / (x_k * N_COINS_U256);
    }
    c = c * d / (ann * N_COINS_U256);

    let b = s_ + d / ann; // b = sum(known) + D/Ann

    // Newton's method: solve y^2 + (b - D)*y = c
    // y_{k+1} = (y_k^2 + c) / (2*y_k + b - D)
    let mut y = d;
    let mut y_prev: U256;

    for _ in 0..MAX_ITERATIONS {
        y_prev = y;
        y = (y * y + c) / (U256::from(2u64) * y + b - d);

        let diff = if y > y_prev { y - y_prev } else { y_prev - y };
        if diff <= U256::ONE {
            return Some(y);
        }
    }

    None // Failed to converge
}
```

use crate::math::*;
use crate::types::*;
use ethnum::U256;

/// Calculate the output amount for a swap, before any token transfers.
/// Returns (output_amount_native, fee_amount_native) in token j's native decimals.
pub fn calc_swap_output(
    i: usize,
    j: usize,
    dx_native: u128,
    balances: &[u128; 3],
    precision_muls: &[u64; 3],
    amp: u64,
    swap_fee_bps: u64,
) -> Result<(u128, u128), ThreePoolError> {
    if dx_native == 0 {
        return Err(ThreePoolError::ZeroAmount);
    }
    if i == j || i >= 3 || j >= 3 {
        return Err(ThreePoolError::InvalidCoinIndex);
    }

    let xp = normalize_all(balances, precision_muls);
    let d = get_d(&xp, amp).ok_or(ThreePoolError::InvariantNotConverged)?;

    // New balance of token i after swap (normalized)
    let new_x_i = xp[i] + normalize_balance(dx_native, precision_muls[i]);

    // Solve for new balance of token j
    let new_y_j = get_y(i, j, new_x_i, &xp, amp, d)
        .ok_or(ThreePoolError::InvariantNotConverged)?;

    // Raw output in normalized (18-dec) terms
    let dy_normalized = xp[j].checked_sub(new_y_j)
        .ok_or(ThreePoolError::InsufficientLiquidity)?;

    // Apply fee in normalized space
    let fee_normalized = dy_normalized * U256::from(swap_fee_bps) / U256::from(10_000u64);
    let dy_after_fee_normalized = dy_normalized - fee_normalized;

    // Convert back to native decimals
    let output = denormalize_balance(dy_after_fee_normalized, precision_muls[j]);
    let fee = denormalize_balance(fee_normalized, precision_muls[j]);

    Ok((output, fee))
}

```rust
use crate::math::*;
use crate::types::*;
use ethnum::U256;

/// Calculate LP tokens to mint for a deposit.
/// Returns (lp_to_mint_u128, fees_native[3]).
///
/// For first deposit: lp_minted = D, no fees.
/// For subsequent: lp_minted proportional to D increase, with imbalance fees.
pub fn calc_add_liquidity(
    amounts: &[u128; 3],
    old_balances: &[u128; 3],
    precision_muls: &[u64; 3],
    lp_total_supply: u128,
    amp: u64,
    fee_bps: u64,
) -> Result<(u128, [u128; 3]), ThreePoolError> {
    let has_deposit = amounts.iter().any(|&a| a > 0);
    if !has_deposit {
        return Err(ThreePoolError::ZeroAmount);
    }

    // New balances after deposit
    let mut new_balances = *old_balances;
    for k in 0..3 {
        new_balances[k] += amounts[k];
    }

    let old_xp = normalize_all(old_balances, precision_muls);
    let new_xp = normalize_all(&new_balances, precision_muls);

    let d0 = if lp_total_supply == 0 {
        U256::ZERO
    } else {
        get_d(&old_xp, amp).ok_or(ThreePoolError::InvariantNotConverged)?
    };
    let d1 = get_d(&new_xp, amp).ok_or(ThreePoolError::InvariantNotConverged)?;

    if d1 <= d0 {
        return Err(ThreePoolError::ZeroAmount);
    }

    // First deposit: mint D1 LP tokens, no fees
    if lp_total_supply == 0 {
        return Ok((d1.as_u128(), [0, 0, 0]));
    }

    // Imbalance fee: fee * N / (4 * (N-1)) applied to deviation from ideal
    let imbalance_fee_bps = fee_bps * 3 / (4 * 2); // N=3: fee*3/8
    let mut fees_native = [0u128; 3];
    let mut adjusted_xp = new_xp;

    for k in 0..3 {
        // Ideal new balance = old_balance * d1/d0
        let ideal = old_xp[k] * d1 / d0;
        let diff = if new_xp[k] > ideal {
            new_xp[k] - ideal
        } else {
            ideal - new_xp[k]
        };
        let fee_k = diff * U256::from(imbalance_fee_bps) / U256::from(10_000u64);
        fees_native[k] = denormalize_balance(fee_k, precision_muls[k]);
        adjusted_xp[k] = new_xp[k] - fee_k;
    }

    let d2 = get_d(&adjusted_xp, amp).ok_or(ThreePoolError::InvariantNotConverged)?;

    // LP minted = total_supply * (d2 - d0) / d0
    let lp_minted = U256::from(lp_total_supply) * (d2 - d0) / d0;

    Ok((lp_minted.as_u128(), fees_native))
}

/// Proportional withdrawal: no fees, returns array of amounts in native decimals.
pub fn calc_remove_liquidity(
    lp_burn: u128,
    balances: &[u128; 3],
    lp_total_supply: u128,
) -> [u128; 3] {
    let mut amounts = [0u128; 3];
    for k in 0..3 {
        amounts[k] = (balances[k] as u128) * lp_burn / lp_total_supply;
    }
    amounts
}

/// Single-token withdrawal: user burns LP and receives one token.
/// Returns (amount_native, fee_native).
pub fn calc_remove_one_coin(
    lp_burn: u128,
    coin_index: usize,
    balances: &[u128; 3],
    precision_muls: &[u64; 3],
    lp_total_supply: u128,
    amp: u64,
    fee_bps: u64,
) -> Result<(u128, u128), ThreePoolError> {
    let xp = normalize_all(balances, precision_muls);
    let d0 = get_d(&xp, amp).ok_or(ThreePoolError::InvariantNotConverged)?;

    // D after burning LP
    let d1 = d0 - U256::from(lp_burn) * d0 / U256::from(lp_total_supply);

    // Calculate new balance of target coin using get_y logic adapted for removal
    // We need to find what xp[coin_index] would be if D = d1 and other balances
    // are reduced proportionally first, then fees applied.
    let new_y = get_y_d(coin_index, &xp, amp, d1)?;

    // Imbalance fee per coin
    let imbalance_fee_bps = fee_bps * 3 / (4 * 2);

    let mut xp_reduced = xp;
    for k in 0..3 {
        let ideal = xp[k] * d1 / d0;
        let diff = if xp[k] > ideal { xp[k] - ideal } else { ideal - xp[k] };
        xp_reduced[k] -= diff * U256::from(imbalance_fee_bps) / U256::from(10_000u64);
    }

    let new_y_reduced = get_y_d(coin_index, &xp_reduced, amp, d1)?;
    let dy = xp_reduced[coin_index] - new_y_reduced;
    let fee = (xp[coin_index] - new_y) - dy;

    Ok((
        denormalize_balance(dy, precision_muls[coin_index]),
        denormalize_balance(fee, precision_muls[coin_index]),
    ))
}

/// Like get_y but solves for balance[j] given a specific D value.
/// All other balances remain at their current values.
pub fn get_y_d(j: usize, xp: &[U256; 3], amp: u64, d: U256) -> Option<U256> {
    let ann = U256::from(amp) * N_COINS_U256;
    let mut c = d;
    let mut s_ = U256::ZERO;

    for k in 0..3 {
        if k == j { continue; }
        s_ += xp[k];
        c = c * d / (xp[k] * N_COINS_U256);
    }
    c = c * d / (ann * N_COINS_U256);
    let b = s_ + d / ann;

    let mut y = d;
    for _ in 0..MAX_ITERATIONS {
        let y_prev = y;
        y = (y * y + c) / (U256::from(2u64) * y + b - d);
        let diff = if y > y_prev { y - y_prev } else { y_prev - y };
        if diff <= U256::ONE {
            return Some(y);
        }
    }
    None
}

        Ok((output, fee, s.config.tokens[i].ledger_id, s.config.tokens[j].ledger_id))

    transfers::transfer_to_user(ledger_j, caller, output)
        .await
        .map_err(|e| ThreePoolError::TransferFailed {
            token: format!("token_{}", j),
            reason: e,
        })?;

    // Update state
    mutate_state(|s| {
        let i = i as usize;
        let j = j as usize;
        s.balances[i] += dx;
        s.balances[j] -= output + fee; // fee stays in pool
        // Track admin fee portion
        let admin_fee = fee * s.config.admin_fee_bps as u128 / 10_000;
        s.admin_fees[j] += admin_fee;
    });

    Ok(output)

#[update]
pub async fn remove_liquidity(
    lp_burn: u128,
    min_amounts: [u128; 3],
) -> Result<[u128; 3], ThreePoolError> {
    let caller = ic_cdk::caller();

    let amounts = read_state(|s| {
        let user_lp = s.lp_balances.get(&caller).copied().unwrap_or(0);
        if lp_burn > user_lp {
            return Err(ThreePoolError::InsufficientLiquidity);
        }
        let amounts = liquidity::calc_remove_liquidity(lp_burn, &s.balances, s.lp_total_supply);
        for k in 0..3 {
            if amounts[k] < min_amounts[k] {
                return Err(ThreePoolError::SlippageExceeded);
            }
        }
        Ok(amounts)
    })?;

    // Transfer tokens from pool to user
    for k in 0..3 {
        if amounts[k] > 0 {
            let ledger = read_state(|s| s.config.tokens[k].ledger_id);
            transfers::transfer_to_user(ledger, caller, amounts[k])
                .await
                .map_err(|e| ThreePoolError::TransferFailed {
                    token: format!("token_{}", k),
                    reason: e,
                })?;
        }
    }

    // Update state
    mutate_state(|s| {
        for k in 0..3 {
            s.balances[k] -= amounts[k];
        }
        let user_lp = s.lp_balances.get_mut(&caller).unwrap();
        *user_lp -= lp_burn;
        if *user_lp == 0 {
            s.lp_balances.remove(&caller);
        }
        s.lp_total_supply -= lp_burn;
    });

    Ok(amounts)
}
```

use crate::state::*;
use crate::types::*;

const MIN_RAMP_TIME: u64 = 86400; // 1 day minimum
const MAX_A_CHANGE: u64 = 10;     // Max 10x change per ramp

pub fn ramp_a(
    future_a: u64,
    future_a_time: u64,
    caller: candid::Principal,
    now: u64,
) -> Result<(), ThreePoolError> {
    read_state(|s| {
        if caller != s.config.admin {
            return Err(ThreePoolError::Unauthorized);
        }
        Ok(())
    })?;

    mutate_state(|s| {
        let current_a = crate::math::get_a(
            s.config.initial_a, s.config.future_a,
            s.config.initial_a_time, s.config.future_a_time, now,
        );

        // Safety checks
        if future_a_time < now + MIN_RAMP_TIME {
            return Err(ThreePoolError::Unauthorized); // ramp too short
        }
        if future_a > current_a * MAX_A_CHANGE || future_a * MAX_A_CHANGE < current_a {
            return Err(ThreePoolError::Unauthorized); // change too large
        }

        s.config.initial_a = current_a;
        s.config.future_a = future_a;
        s.config.initial_a_time = now;
        s.config.future_a_time = future_a_time;
        Ok(())
    })
}
